var titleFinder = ruleset(
    // Give any title tag a score of 1, and tag it as title-ish:
    rule(dom("title"),
         node => [{score: 1, flavor: 'titley'}]),

    // Give any OpenGraph meta tag a score of 2,
    // and tag it as title-ish as well:
    rule(dom('meta[property="og:title"]'),
         node => [{score: 2, flavor: 'titley'}]),

    // Take all title-ish things, and punish them if they contain
    // navigational claptrap like colons or dashes:
    rule(flavor("titley"),
         node => [{score: hasColonsOrDashes(node.element) ? .5 : 1}])
);
